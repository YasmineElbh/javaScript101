// += operator
let score = 100;
// ==>
score += 100;
console.log(score);

// -= operator
let score = 100;
// ==>
score -= 100;
console.log(score);

// *= operator
let score = 100;
// ==>
score *= 100;
console.log(score);


// /= operator
let score = 100;
// ==>
score /= 100;
console.log(score);

// %= operator
let score = 100;
// ==>
score %= 100;
console.log(score);
